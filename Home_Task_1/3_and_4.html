• let mode = "normal"; - нормальное выражение
  • let my_valu3 = 102; - нормальное выражение
  • let 3my_value3 = "102"; - ненормальное выражение (имя переменной не может начинаться с цифры *существенно*, да и число превратилось в строку *несущественно*)
  • let __hello__ = "world"; - нормальное выражение, хотя имя переменной начинается с двух знаков пробела. Я бы такое имя не задавал
  • let $$$ = "money"; - нормальное выражение, хотя имя переменной неудобное
  • let !0_world = true; - ненормальное выражение, т.к. с ! имя переменной не может начинаться
